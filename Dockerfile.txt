# Use official .NET runtime as base image
FROM mcr.microsoft.com/dotnet/aspnet:6.0 AS base
WORKDIR /app
EXPOSE 80
EXPOSE 443

# Build stage
FROM mcr.microsoft.com/dotnet/sdk:6.0 AS build
WORKDIR /src

# Copy solution and restore
COPY ["OyoAgro.Api/OyoAgro.Api.csproj", "OyoAgro.Api/"]
RUN dotnet restore "OyoAgro.Api/OyoAgro.Api.csproj"

# Copy everything and build
COPY . .
WORKDIR "/src/OyoAgro.Api"
RUN dotnet build "OyoAgro.Api.csproj" -c Release -o /app/build

# Publish stage
FROM build AS publish
RUN dotnet publish "OyoAgro.Api.csproj" -c Release -o /app/publish

# Final runtime image
FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "OyoAgro.Api.dll"]
